Vanilla-Tilt 立體傾斜效果

套件官網

範例程式碼


<!-- Vanilla-Tilt -->
<script src="https://unpkg.com/vanilla-tilt@1.7.0/dist/vanilla-tilt.min.js"></script>

<!-- 頁面內容，包含一個寬 300px 高 300px 的元素 -->
<div id="tiltElement" style="width: 300px; height: 300px; background-color: lightcoral;">
    Hover to tilt this element.
</div>

<!-- 引入 VanillaTilt.js 並初始化傾斜效果 -->
<script>
    VanillaTilt.init(document.querySelector("#tiltElement"), {
        max: 25,
        speed: 400
    });
</script>

效果展示

Hover to tilt this element.
