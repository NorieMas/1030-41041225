套件官網

範例程式碼

        
<!-- 引入 Swiper 的 CSS 文件 -->
<link href="https://unpkg.com/swiper/swiper-bundle.min.css" rel="stylesheet">
        
<!-- 初始化 Swiper 內容 -->
<div class="swiper-container">
    <div class="swiper-wrapper">
        <div class="swiper-slide"><img src="https://via.placeholder.com/400x300/ff7f7f" alt="Image 1"></div>
        <div class="swiper-slide"><img src="https://via.placeholder.com/400x300/7fff7f" alt="Image 2"></div>
        <div class="swiper-slide"><img src="https://via.placeholder.com/400x300/7f7fff" alt="Image 3"></div>
    </div>
    <div class="swiper-pagination"></div>
</div>
        
<!-- 引入 Swiper 的 JS 文件 -->
<script src="https://unpkg.com/swiper/swiper-bundle.min.js"></script>
        
<!-- 初始化 Swiper 並設置參數 -->
<script>
    var swiper = new Swiper('.swiper-container', {
        loop: true,
        pagination: {
            el: '.swiper-pagination',
            clickable: true,
        },
    });
</script>
