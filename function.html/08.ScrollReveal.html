ScrollReveal 滾動顯示動畫

套件官網

範例程式碼


<!-- ScrollReveal -->
<script src="https://unpkg.com/scrollreveal"></script>
<!-- 頁面內容，包含一個高度為 100vh 的元素 -->
<div style="height: 100vh;">Scroll Down</div>

<div id="revealElement" style="height: 200px; background-color: lightblue;">
    This content will animate when it scrolls into view.
</div>

<!-- 引入 ScrollReveal 動畫效果 -->
<script>
    ScrollReveal().reveal('#revealElement', { 
        distance: '50px', 
        duration: 1000, 
        easing: 'ease-in-out',
        origin: 'bottom' 
    });
</script>
