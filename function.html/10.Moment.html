Moment.js 時間處理

套件官網

範例程式碼


<!-- Moment.js -->
<script src="https://cdn.jsdelivr.net/npm/moment@2.29.1/moment.min.js"></script>

<!-- 顯示當前時間的元素 -->
<p>Current time: <span id="currentTime"></span></p>

<!-- 使用 Moment.js 顯示當前時間 -->
<script>
    document.getElementById('currentTime').innerText = moment().format('YYYY-MM-DD HH:mm:ss');
</script>
